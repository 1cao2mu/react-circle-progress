### Installation

```cmake
npm i react-circle-progress
```

### Usage

```react
import { SvgCirPro,CanCirPro,CircleProgress } from "react-circle-progress";
```


### More examples

```react
ReactDOM.render(<div>

    <CircleProgress percent={33} />

    <CanCirPro percent={88} />

    <SvgCirPro percent={100}/>

</div>, document.getElementById('root'));

```
